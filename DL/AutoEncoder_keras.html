from keras import layers, models  # (Input, Dense), (Model)   #  import

class AE(models.Model):                                       # AE   
    def __init__(self, x_nodes=784, z_dim=36):                # x_nodes :  , z_dim :   
        x_shape = (x_nodes,)                                  # x_shape   
        x = layers.Input(shape=x_shape)                       #  
        z = layers.Dense(z_dim, activation='relu')(x)         #   z_dim   ,   'relu'
        y = layers.Dense(x_nodes, activation='sigmoid')(z)    #  ,   'sigmoid' (<-    ,     )
        
        #      
        super().__init__(x, y)                                

        self.x = x                #                              
        self.z = z                #  
        self.z_dim = z_dim        #    

        # AE    
        self.compile(optimizer = 'adam',
                     #optimizer='adadelta',        #   : adadelta, (+ (   ) = adam)
                     loss = 'mean_squared_error',
                     #loss='binary_crossentropy',  # loss = (  ) (MSE)
                     metrics=['accuracy'])        #  : 
        

    # AE           
    def Encoder(self):
        return models.Model(self.x, self.z)

    #          
    def Decoder(self):
        z_shape = (self.z_dim,)
        z = layers.Input(shape=z_shape)       #      
        y_layer = self.layers[-1]             #     self , layers[-1]   
        y = y_layer(z)
        return models.Model(z, y)

import matplotlib.pyplot as plt

import os

#      history       

def plot_acc(history, title=None):        # Accuracy() Visualization
    # summarize history for accuracy
    if not isinstance(history, dict):
        history = history.history

    plt.plot(history['acc'])        # accuracy
    plt.plot(history['val_acc'])    # validation accuracy
    if title is not None:
        plt.title(title)
    plt.ylabel('Accracy')
    plt.xlabel('Epoch')
    plt.legend(['Training data', 'Validation data'], loc=0)
    # plt.show()

def show_ae(autoencoder):

    #     
    encoder = autoencoder.Encoder()
    decoder = autoencoder.Decoder()

    # test  ,   
    encoded_imgs = encoder.predict(X_test)
    decoded_imgs = decoder.predict(encoded_imgs)

    #    
    n = 10                                  #   
    plt.figure(figsize=(20, 6))             #   

    for i in range(n):

        #   ( , x_test)
        ax = plt.subplot(3, n, i + 1)
        plt.imshow(X_test[i].reshape(28, 28))
        plt.gray()
        ax.get_xaxis().set_visible(False)   #    
        ax.get_yaxis().set_visible(False)

        #   : 2  1     1   (x : 36 )
        ax = plt.subplot(3, n, i + 1 + n)
        plt.stem(encoded_imgs[i].reshape(-1))
        plt.gray()
        ax.get_xaxis().set_visible(False)
        ax.get_yaxis().set_visible(False)

        #  (  )
        ax = plt.subplot(3, n, i + 1 + n + n)
        plt.imshow(decoded_imgs[i].reshape(28, 28))
        plt.gray()
        ax.get_xaxis().set_visible(False)
        ax.get_yaxis().set_visible(False)

    plt.show()

def main():             # AE main  
    
    # AE   
    x_nodes = 784       
    z_dim = 36          #  784 36  

    # AE  
    autoencoder = AE(x_nodes, z_dim)

    #  
    history = autoencoder.fit(X_train, X_train,       # ,   
                              epochs=10,              # 10 
                              batch_size=256,         # 1   256   
                              shuffle=True, 
                              validation_data=(X_test, X_test)) #   


    # Visualization
    plot_acc(history, '(a) Accuracy')  #      
    plt.show()
    plot_loss(history, '(b) Loss')     #      
    plt.show()

    # AE 
    show_ae(autoencoder)
    plt.show()
