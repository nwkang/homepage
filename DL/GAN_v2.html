X = tf.placeholder(tf.float32, [None, n_input]) #        
Z = tf.placeholder(tf.float32, [None, n_noise]) #  Z  

# (Generator)   
G_W1 = tf.Variable(tf.random_normal([n_noise, n_hidden], stddev=0.01)) #   weight 
G_b1 = tf.Variable(tf.zeros([n_hidden])) # 0 
G_W2 = tf.Variable(tf.random_normal([n_hidden, n_input], stddev=0.01))
G_b2 = tf.Variable(tf.zeros([n_input]))

# (Discriminator)   
D_W1 = tf.Variable(tf.random_normal([n_input, n_hidden], stddev=0.01))
D_b1 = tf.Variable(tf.zeros([n_hidden]))

#          
D_W2 = tf.Variable(tf.random_normal([n_hidden, 1], stddev=0.01))
D_b2 = tf.Variable(tf.zeros([1])) #    =   0
In [4]:
# (Generator)  
def generator(noise_z):
    hidden = tf.nn.relu(tf.matmul(noise_z, G_W1) + G_b1) # hidden layer relu . noise_z input .
    output = tf.nn.sigmoid(tf.matmul(hidden, G_W2) + G_b2) # output sigmoid  0 1  output  .
    return output
In [5]:
# (Discriminator)  
def discriminator(inputs):
    hidden = tf.nn.relu(tf.matmul(inputs, D_W1) + D_b1) # inputs() input 
    output = tf.nn.sigmoid(tf.matmul(hidden, D_W2) + D_b2) #  sigmoid  0 1   / 
    return output

def get_noise(batch_size, n_noise):
    return np.random.normal(size=(batch_size, n_noise))

# Discriminator loss 
# (D)   tf.log(D_real)     tf.log(1 - D_gene)     
loss_D = tf.reduce_mean(tf.log(D_real) + tf.log(1 - D_gene))

# Generator loss 
# (G)     D    D_gene    
loss_G = tf.reduce_mean(tf.log(D_gene))
