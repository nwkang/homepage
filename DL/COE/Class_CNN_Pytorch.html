class CNN_Model(nn.Module):
    def __init__(self):
        super(CNN_Model, self).__init__()

        self.conv1 = nn.Conv2d(in_channels=1, out_channels=32, kernel_size=3, padding=1)   # initial input channel of grayscale = 1 -> output channel = # of convolution filter
        self.conv2 = nn.Conv2d(in_channels=32, out_channels=64, kernel_size=3, padding=1)
        self.conv3 = nn.Conv2d(in_channels=64, out_channels=128, kernel_size=3, padding=1)
        self.relu = nn.ReLU() # relu activation function
        self.maxpool = nn.MaxPool2d(kernel_size=2) # Max-pooling

        self.flatten = nn.Flatten()
        self.fc1 = nn.Linear(128*7*7, 256) # 128 is the number of output channels of the previous convolution layer(self.conv3).
                                           # We use 3 times Maxpooling, so image dimension is changed to 56*56 -> 28*28 -> 14*14 -> 7*7
                                           # There are 128 feature maps, each 7x7 in size. Using nn.Flatten(), they are all lined up to create one long vector.
                                           # 256 dimensionality of the output space

        self.fc2 = nn.Linear(256,1) # 1 dimensionality of the output space(predict compliance value)

    def forward(self, x):

        x = self.conv1(x)
        x = self.relu(x)
        x = self.maxpool(x)

        x = self.conv2(x)
        x = self.relu(x)
        x = self.maxpool(x)

        x = self.conv3(x)
        x = self.relu(x)
        x = self.maxpool(x)

        x = self.flatten(x)

        x = self.fc1(x)
        x = self.relu(x)

        x = self.fc2(x)

        return x

7. Model training¶

In [ ]:
# Create DataLoader for training and validation data
train_dataset = TensorDataset(train_images, train_labels)
train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)

# loss function = MSE : mean square error
criterion = nn.MSELoss()

# optimizer = Adam
optimizer = optim.Adam(model.parameters(), lr=learning_rate)

loss_train_history = []
rmse_train_history = []
mape_train_history = []

loss_valid_history = []
rmse_valid_history = []
mape_valid_history = []

for epoch in range(training_epochs):

    # train
    model.train()
    total_loss = 0.0
    rmse_train = 0.0
    mape_train = 0.0
    for images, labels in train_loader:
        optimizer.zero_grad()               ## gradient reset
        outputs = model(images).to('cuda')
        labels = labels.to('cuda')
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        total_loss += loss.item()
        rmse_train += mean_squared_error(outputs, labels, squared=False).item() # returns RMSE value if set to False
        mape_train += mean_absolute_percentage_error(outputs, labels).item() * 100

    average_loss_train = total_loss / len(train_loader)
    average_rmse_train = rmse_train / len(train_loader)
    average_mape_train = mape_train / len(train_loader)

    loss_train_history.append(average_loss_train)
    rmse_train_history.append(average_rmse_train)
    mape_train_history.append(average_mape_train)

    # evaluation
    model.eval()
    with torch.no_grad():
        test_outputs = model(test_images).to('cuda')
        test_labels = test_labels.to('cuda')
        loss_val = criterion(test_outputs, test_labels).item()
        rmse_val = mean_squared_error(test_outputs, test_labels, squared=False).item() # returns RMSE value if set to False
        mape_val = mean_absolute_percentage_error(test_outputs, test_labels).item() * 100

    loss_valid_history.append(loss_val)
    rmse_valid_history.append(rmse_val)
    mape_valid_history.append(mape_val)

    print(f"Epoch [{epoch+1}/{training_epochs}], TRAIN_RMSE: {average_rmse_train:.4f}, TRAIN MAPE: {average_mape_train:.4f}, VAL_RMSE: {rmse_val:.4f}, VAL_MAPE: {mape_val:.4f}")

# Save the model
model_scripted = torch.jit.script(model)
model_scripted.save('CNN_TEST.pth')

def plot_rmse(rmse_train_history, rmse_valid_history, title=None):

    plt.plot(rmse_train_history)
    plt.plot(rmse_valid_history)
    if title is not None:
        plt.title(title)
    plt.ylabel('RMSE')
    plt.xlabel('Epoch')
    plt.legend(['Training data', 'Validation data'], loc=0)

def r_val(true,pred):  # R : Coefficient of determination

    true = pd.DataFrame(true)
    pred = pd.DataFrame(pred)
