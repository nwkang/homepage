# 압축해제된 데이터 경로를 찾아 복사해서 붙여넣어주세요
src = './data_3000/'

# src 경로에 있는 파일 명을 저장합니다. 
files = os.listdir(src) # anomaly와 데이터 수를 맞추기 위해 1000개만

X_norm = []

# 경로와 파일명을 입력으로 넣어 확인하고 
# 데이터를 255로 나눠서 0~1사이로 정규화 하여 X 리스트에 넣습니다. 
for file in files: 
    X_norm.append(img_read(src,file)/255.)

# array로 데이터 변환
X_norm = np.array(X_norm)
y_norm = np.zeros(len(X_norm))

print('Normal shape:',np.shape(X_norm))
print('Y shape:', y_norm.shape)

Y shape: (3000,)
