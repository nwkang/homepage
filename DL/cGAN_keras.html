from __future__ import print_function, division


from keras.datasets import mnist
from keras.layers import Input, Dense, Reshape, Flatten, Dropout, multiply
from keras.layers import BatchNormalization, Activation, ZeroPadding2D, Embedding
from keras.layers.advanced_activations import LeakyReLU
from keras.layers.convolutional import UpSampling2D, Conv2D
from keras.models import Sequential, Model
from keras.optimizers import Adam

import matplotlib.pyplot as plt
import sys
import numpy as np


# ---------------------
#   , model build, compile
# ---------------------

class cGAN():
    def __init__(self):
        # input shape . (mnist  )
        self.img_rows = 28
        self.img_cols = 28
        self.channels = 1
        self.img_shape = (self.img_rows, self.img_cols, self.channels)
        # 0 9 10   
        self.num_classes = 10
        #  dimension
        self.latent_dim = 100
        
        optimizer = Adam(0.0002, 0.5)
        
        # discriminator   (loss optimizer , accaracy )
        self.discriminator = self.build_discriminator()
        self.discriminator.compile(loss='binary_crossentropy',
                                  optimizer = optimizer,
                                  metrics = ['accuracy'])
        
        # generator  
        self.generator = self.build_generator()
        
        # generator noise, (0 9) .(   )
        noise = Input(shape=(self.latent_dim,))
        label = Input(shape=(1,))
        img = self.generator([noise, label])
        
        #       discriminator  .  generator 
        self.discriminator.trainable = False
        
        # discriminator       
        validity = self.discriminator([img, label])
        
        # generator discriminator  . 
        # (      )
        # discriminator  generator 
        self.combined = Model([noise, label], validity)
        self.combined.compile(loss='binary_crossentropy', optimizer=optimizer)
        
        
    # ---------------------
    #  Generator   (input : noise, label / output : image)
    # ---------------------    
        
    def build_generator(self):
        
        model = Sequential()
        
        model.add(Dense(256, input_dim=self.latent_dim))
        model.add(LeakyReLU(alpha=0.2))
        model.add(BatchNormalization(momentum=0.8))
        model.add(Dense(512))
        model.add(LeakyReLU(alpha=0.2))
        model.add(BatchNormalization(momentum=0.8))
        model.add(Dense(1024))
        model.add(LeakyReLU(alpha=0.2))
        model.add(BatchNormalization(momentum=0.8))
        model.add(Dense(np.prod(self.img_shape), activation='tanh'))
        model.add(Reshape(self.img_shape))
                  
        model.summary()
        
        # noise, label  
        noise = Input(shape=(self.latent_dim,))
        label = Input(shape=(1,), dtype='int32')
        # label noise  
        label_embedding = Flatten()(Embedding(self.num_classes, self.latent_dim)(label))
        
        # noise label () -> input 
        model_input = multiply([noise, label_embedding])
        img = model(model_input)
        
        return Model([noise, label], img)
    
    
    # ---------------------
    #  Discriminator   (input : image, label / output : (0 1 ))
    # ---------------------    
    
    def build_discriminator(self):
        
        model = Sequential()
        
        model.add(Dense(512, input_dim=np.prod(self.img_shape)))
        model.add(LeakyReLU(alpha=0.2))
        model.add(Dense(512))
        model.add(LeakyReLU(alpha=0.2))
        model.add(Dropout(0.4))
        model.add(Dense(256))
        model.add(LeakyReLU(alpha=0.2))
        model.add(Dropout(0.4))
        model.add(Dense(1, activation='sigmoid'))
        model.summary()
        
        # noise, label  
        img = Input(shape=self.img_shape)
        label = Input(shape=(1,), dtype='int32')
                       
        # (10, (28x28))     label  Flatten
        label_embedding = Flatten()(Embedding(self.num_classes, np.prod(self.img_shape))(label))
        #  (Flatten)
        flat_img = Flatten()(img)
                       
        #   label embedding 
        model_input = multiply([flat_img, label_embedding])
        
        #  input   
        validity = model(model_input)
        
        return Model([img, label], validity)
    
    
    # ---------------------
    #    
    # ---------------------    
    
    def train(self, epochs, batch_size=128, sample_interval=50):
        
        # mnist   
        (X_train, y_train), (__, __) = mnist.load_data()
        
        #  (Rescale -1 to 1), input shape(60000, 28, 28, 1) , label shape   
        X_train = X_train / 127.5 - 1.
        X_train = np.expand_dims(X_train, axis=3)
        y_train = y_train.reshape(-1, 1)
        
        #   . valid 1, fake 0
        valid = np.ones((batch_size, 1))
        fake = np.zeros((batch_size, 1))
        
        for epoch in range(epochs):
            
            # ---------------------
            #  Train Discriminator
            # ---------------------
            
            # batch_size    
            idx = np.random.randint(0, X_train.shape[0], batch_size)
            imgs, labels = X_train[idx], y_train[idx]
            
            # Sample noise (batch_size), noise generator   
            noise = np.random.normal(0, 1, (batch_size, self.latent_dim))
            gen_imgs = self.generator.predict([noise, labels])
            
            # discriminator .   1 ,   0 
            # discriminator    
            # valid fake    binary_crossentropy  .
            d_loss_real = self.discriminator.train_on_batch([imgs, labels], valid)
            d_loss_fake = self.discriminator.train_on_batch([gen_imgs, labels], fake)
            
            # real   fake   discriminator loss accracy  .
            d_loss = 0.5 * np.add(d_loss_real, d_loss_fake)
            
            # ---------------------
            #  Train Generator
            # ---------------------
            
            #   
            sampled_labels = np.random.normal(0, 10, batch_size).reshape(-1, 1)
            
            # generator . noise   generator   discriminator 1   
            g_loss = self.combined.train_on_batch([noise, sampled_labels], valid)
            
            # 1000 epoch loss accracy 
            if epoch % 1000 == 0:
                print("%d [D loss: %f, acc.: %.2f%%] [G loss: %f]" % (epoch, d_loss[0], 100*d_loss[1], g_loss))
            
            # save_interval(5000) generator   
            if epoch % sample_interval == 0:    
                self.sample_images(epoch)
                
    #  
    def sample_images(self, epoch):
        r, c = 2, 5
        noise = np.random.normal(0, 1, (r*c, 100))
        sampled_labels = np.arange(0, 10).reshape(-1, 1)
                       
        gen_imgs = self.generator.predict([noise, sampled_labels])
                       
        # Rescale images 0 - 1
        gen_imgs = 0.5 * gen_imgs + 0.5
                       
        fig, axs = plt.subplots(r,c)
        cnt = 0
        for i in range(r):
            for j in range(c):
                axs[i,j].imshow(gen_imgs[cnt,:,:,0], cmap='gray')
                axs[i,j].set_title("Digit: %d" %sampled_labels[cnt])
                axs[i,j].axis('off')
                cnt += 1
